const APP_VERSION = "1.3";

const STORAGE_KEY = "savfit_premium_state_v13";

const CURRENT_KEY = "savfit_current_workout_v13";

const LAST_SCREEN_KEY = "savfit_last_screen_v13";

const $ = id => document.getElementById(id);

const today = () => new Date().toISOString().slice(0,10);

function defaultState(){

  return {

    profile:{

      name:"",

      avatar:""

    },

    theme:{

      bg:"#000000",

      card:"#101010",

      accent:"#ff6600",

      text:"#ffffff"

    },

    settings:{

      restSeconds:60,

      vibration:false

    },

    logs:[],

    meals:[],

    checkins:[],

    goals:[],

    achievements:[],

    customExercises:[],

    macroTargets:{

      calories:0,

      protein:0

    }

  };

}

let state = safeLoadState();

let currentScreen = "home";

let navHistory = ["home"];

let currentWorkout = safeLoadCurrentWorkout();

function safeLoadState(){

  try{

    const raw = localStorage.getItem(STORAGE_KEY);

    if(!raw) return defaultState();

    const loaded = JSON.parse(raw);

    const base = defaultState();

    return {

      ...base,

      ...loaded,

      profile:{...base.profile,...(loaded.profile||{})},

      theme:{...base.theme,...(loaded.theme||{})},

      settings:{...base.settings,...(loaded.settings||{})},

      macroTargets:{...base.macroTargets,...(loaded.macroTargets||{})},

      logs:Array.isArray(loaded.logs)?loaded.logs:[],

      meals:Array.isArray(loaded.meals)?loaded.meals:[],

      checkins:Array.isArray(loaded.checkins)?loaded.checkins:[],

      goals:Array.isArray(loaded.goals)?loaded.goals:[],

      achievements:Array.isArray(loaded.achievements)?loaded.achievements:[],

      customExercises:Array.isArray(loaded.customExercises)?loaded.customExercises:[]

    };

  }catch(e){

    return defaultState();

  }

}

function safeLoadCurrentWorkout(){

  try{

    const raw = localStorage.getItem(CURRENT_KEY);

    if(!raw){

      return {

        date:today(),

        focus:"",

        data:[]

      };

    }

    const loaded = JSON.parse(raw);

    return {

      date:loaded.date || today(),

      focus:loaded.focus || "",

      data:Array.isArray(loaded.data)?loaded.data:[]

    };

  }catch(e){

    return {

      date:today(),

      focus:"",

      data:[]

    };

  }

}

function saveState(){

  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

}

  localStorage.setItem(CURRENT_KEY, JSON.stringify(currentWorkout));

}

function clearStoredCurrentWorkout(){

  localStorage.removeItem(CURRENT_KEY);

}

function showScreen(id, push=true){

  const target = $(id);

  if(!target) return;

  document.querySelectorAll(".screen").forEach(screen=>{

    screen.classList.remove("active");

  });

  target.classList.add("active");

  currentScreen = id;

  if(push){

    const last = navHistory[navHistory.length-1];

    if(last !== id) navHistory.push(id);

  }

  saveLastScreen(id);

  updateTopTitle(id);

  updateTabs(id);

  renderScreen(id);

}

function goBack(){

  if(navHistory.length <= 1){

    showScreen("home");

    return;

  }

  navHistory.pop();

  const previous = navHistory.pop() || "home";

  showScreen(previous);

}

    workout:"Workout",

    progress:"Progress",

    macros:"Macros",

    goals:"Goals",

    stats:"Stats",

    stretches:"Stretches",

    settings:"Settings",

    backup:"Backup",

    help:"Help"

  };

  $("topTitle").innerText = (titles[id] || "SavFit") + " — 1.3";

}

function updateTabs(id){

  ["home","workout","log","progress","macros"].forEach(tab=>{

    const el = $("tab-"+tab);

    if(el) el.classList.toggle("active-tab", tab===id);

  });

}

  document.documentElement.style.setProperty("--card", state.theme.card || "#101010");

  document.documentElement.style.setProperty("--accent", state.theme.accent || "#ff6600");

  document.documentElement.style.setProperty("--text", state.theme.text || "#ffffff");

}

function saveTheme(){

  state.theme.bg = $("bgPicker").value || "#000000";

  state.theme.accent = $("accentPicker").value || "#ff6600";

  state.theme.card = state.theme.bg === "#ffffff" ? "#f1f1f1" : "#101010";

  state.theme.text = state.theme.bg === "#ffffff" ? "#000000" : "#ffffff";

  saveState();

  applyTheme();

  popup("Theme saved");

  renderScreen(currentScreen);

}

function uploadAvatar(event){

  const file = event.target.files && event.target.files[0];

  if(!file) return;

  if(file.size > 2.5 * 1024 * 1024){

    popup("Image is too large. Choose a smaller picture.");

    return;

  }

  const reader = new FileReader();

  reader.onload = () => {

    state.profile.avatar = reader.result;

    saveState();

    applyProfileToUI();

    popup("Avatar saved. Cartoonify will be added in a future online update.");

  };

  reader.onerror = () => popup("Could not load image.");

  reader.readAsDataURL(file);

}

function daysAgo(date){

  return Math.floor((new Date(today()) - new Date(date)) / 86400000);

}

function calculateStreak(){

  if(state.logs.length===0) return 0;

  const dates = [...new Set(state.logs.map(l=>l.date))].sort().reverse();

  let streak = 0;

  let check = new Date(today());

  for(let i=0;i<dates.length;i++){

    const d = dates[i];

    const expected = check.toISOString().slice(0,10);

    if(d===expected){

      streak++;

      check.setDate(check.getDate()-1);

    }else if(i===0 && daysAgo(d)===1){

      streak++;

      check = new Date(d);

      check.setDate(check.getDate()-1);

    }else{

      break;

    }

  }

  return streak;

}

function clamp(n,min,max){

  return Math.max(min,Math.min(max,n));

}

function buildRepInputs(select, existingReps=null){

  const card = select.closest(".exercise-card");

  const box = card.querySelector(".repsBox");

  const count = Number(select.value);

  const old = existingReps || [...box.querySelectorAll(".repInput")].map(i=>i.value);

  box.innerHTML = "";

  for(let i=0;i<count;i++){

    const input = document.createElement("input");

    input.className = "repInput";

    input.type = "number";

    input.placeholder = `Set ${i+1} reps`;

    input.value = old[i] || "";

    box.appendChild(input);

  }

}

function getLastWeight(name){

  for(let i=state.logs.length-1;i>=0;i--){

    const found = state.logs[i].data.find(e=>e.name===name);

    if(found && found.weight) return found.weight;

  }

  return "";

}

function progressionSuggestion(name){

  const entries = [];

  state.logs.forEach(log=>{

    const found = log.data.find(e=>e.name===name);

    if(found){

      entries.push({

        weight:Number(found.weight)||0,

        reps:(found.reps||[]).reduce((a,b)=>a+(Number(b)||0),0),

        form:Number(found.form)||3,

        difficulty:Number(found.difficulty)||3

      });

    }

  });

  if(entries.length < 2) return "Progression: log this exercise more times for smart suggestions.";

  const last = entries[entries.length-1];

  const prev = entries[entries.length-2];

  if(last.weight > prev.weight){

    return "Progression: weight increased. Maintain form before adding more.";

  }

  if(last.weight === prev.weight && last.reps > prev.reps && last.form >= 4){

    return "Progression: reps improved with good form. Consider a small increase next time.";

  }

  if(last.form <= 2){

    return "Progression: keep weight steady and prioritise form.";

  }

  return "Progression: maintain or add a small increase if form feels strong.";

}

function showExerciseNotes(name){

  const real = unsafedText(name);

  const notes = exerciseNotes[real] || ["No notes saved for this exercise yet."];

  popup(notes.map(n=>"• "+n).join("\n"),real);

}

function safeText(text){

  return String(text).replace(/'/g,"__APOS__").replace(/"/g,"__QUOTE__");

}

function unsafedText(text){

  return String(text).replace(/__APOS__/g,"'").replace(/__QUOTE__/g,'"');

}

function checkPersonalBest(entry){

  const previous = [];

  state.logs.forEach(log=>{

    log.data.forEach(e=>{

      if(e.name===entry.name) previous.push(Number(e.weight)||0);

    });

  });

  const best = previous.length ? Math.max(...previous) : 0;

  const current = Number(entry.weight)||0;

  if(current > best && best > 0){

    unlockAchievement("New PR");

    popup(`New PR!\n${entry.name}: ${current}kg 🎉`);

  }

}

function unlockWorkoutAchievements(){

  if(state.logs.length >= 1) unlockAchievement("First Workout");

  if(state.logs.length >= 3) unlockAchievement("3 Workouts");

  if(state.logs.length >= 7) unlockAchievement("7 Workouts");

  if(state.logs.length >= 30) unlockAchievement("30 Workouts");

  const volume = calculateTotalVolume();

  if(volume >= 10000) unlockAchievement("10,000kg Club");

}

}

let restTimerInterval = null;

function startRestTimer(){

  clearInterval(restTimerInterval);

  let seconds = Number(state.settings.restSeconds)||60;

  const bubble = $("timerBubble");

  bubble.style.display = "flex";

  bubble.innerText = seconds;

  restTimerInterval = setInterval(()=>{

    seconds--;

    bubble.innerText = seconds;

    if(seconds <= 0){

      clearInterval(restTimerInterval);

      restTimerInterval = null;

      bubble.style.display = "none";

      vibrate([250,120,250,120,400]);

      popup("Rest complete.");

    }

function clearProgressCanvas(){

  ctx.clearRect(0,0,canvas.width,canvas.height);

function drawProgress(){

  const name = $("progressExercise").value;

  if(!name || name==="No exercise data yet") return;

  const rows = [];

  state.logs.forEach(log=>{

    const ex = log.data.find(e=>e.name===name);

    if(ex){

      rows.push({

        date:log.date,

        weight:Number(ex.weight)||0,

        reps:(ex.reps||[]).join("/"),

        difficulty:ex.difficulty || "-",

        form:ex.form || "-"

      });

    }

  });

  const canvas = $("progressCanvas");

  const ctx = canvas.getContext("2d");

  ctx.clearRect(0,0,canvas.width,canvas.height);

  if(rows.length===0){

    $("progressInfo").innerHTML = `<div class="card empty">No data for this exercise.</div>`;

    return;

  }

  const pad = 34;

  const w = canvas.width;

  const h = canvas.height;

  const maxWeight = Math.max(...rows.map(r=>r.weight),1);

  ctx.strokeStyle = state.theme.accent || "#ff6600";

  ctx.fillStyle = "#fff";

  ctx.lineWidth = 2;

  ctx.font = "11px Arial";

  // axes

  ctx.beginPath();

  ctx.moveTo(pad,pad);

  ctx.lineTo(pad,h-pad);

  ctx.lineTo(w-pad,h-pad);

  ctx.stroke();

  // labels

  ctx.fillStyle = "#fff";

  ctx.fillText("kg",6,20);

  ctx.fillText("date",w-45,h-8);

  const points = rows.map((r,i)=>{

    const x = pad + (i * ((w - pad*2) / Math.max(rows.length-1,1)));

    const y = (h-pad) - ((r.weight / maxWeight) * (h-pad*2));

    return {x,y,row:r};

  });

  // line

  ctx.beginPath();

  points.forEach((p,i)=>{

    if(i===0) ctx.moveTo(p.x,p.y);

    else ctx.lineTo(p.x,p.y);

  });

  ctx.stroke();

  // points

  points.forEach(p=>{

    ctx.beginPath();

    ctx.arc(p.x,p.y,5,0,Math.PI*2);

    ctx.fillStyle = state.theme.accent || "#ff6600";

    ctx.fill();

  });

  // y max label

  ctx.fillStyle = "#fff";

  ctx.fillText(maxWeight+"kg",4,pad+4);

  // x labels sparse

  points.forEach((p,i)=>{

    if(i===0 || i===points.length-1){

      ctx.fillText(p.row.date.slice(5),p.x-20,h-16);

    }

  });

  canvas.onclick = function(evt){

    const rect = canvas.getBoundingClientRect();

    const x = (evt.clientX - rect.left) * (canvas.width / rect.width);

    const y = (evt.clientY - rect.top) * (canvas.height / rect.height);

    let nearest = null;

    let dist = Infinity;

    points.forEach(p=>{

      const d = Math.hypot(p.x-x,p.y-y);

      if(d < dist){

        dist = d;

        nearest = p;

      }

    });

    if(nearest && dist < 25){

      const r = nearest.row;

      popup(`${name}\nDate: ${r.date}\nWeight: ${r.weight}kg\nReps: ${r.reps}\nDifficulty: ${r.difficulty}\nForm: ${r.form}`,"Progress Detail");

    }

  };

  const best = rows.reduce((a,b)=>b.weight>a.weight?b:a,rows[0]);

  const last = rows[rows.length-1];

  $("progressInfo").innerHTML = `

    <div class="card">

      <h3>${name}</h3>

      <p>Best: ${best.weight}kg on ${best.date}</p>

      <p>Latest: ${last.weight}kg on ${last.date}</p>

    </div>

  `;

}

      <p>Protein: ${protein}g/${proteinTarget || "-"}g</p>

      <div class="progress-wrap"><div class="progress-bar" style="width:${proteinPercent}%"></div></div>

      <p class="small">${calTarget ? Math.max(0,calTarget-calories)+" calories remaining" : ""}</p>

function makeCSV(){

  let csv = "Date,Focus,Exercise,Weight,Reps,Difficulty,Form\n";

  state.logs.forEach(log=>{

    log.data.forEach(e=>{

      const safeName = String(e.name || "").replace(/"/g,'""');

      csv += `${log.date},${log.focus},"${safeName}",${e.weight || ""},"${(e.reps||[]).join("/")}",${e.difficulty || ""},${e.form || ""}\n`;

    });

  });

  return csv;

}

function copyCSV(){

  if(state.logs.length===0){

    popup("No workout data to copy.");

    return;

  }

  const csv = makeCSV();

  if(navigator.clipboard && navigator.clipboard.writeText){

    navigator.clipboard.writeText(csv)

      .then(()=>popup("CSV copied to clipboard."))

      .catch(()=>showCSVText(csv));

  }else{

    showCSVText(csv);

  }

}

function showCSVText(csv){

  $("popupTitle").innerText = "CSV Export";

  $("popupMessage").innerText = csv;

  $("popup").style.display = "flex";

}

function downloadCSV(){

  if(state.logs.length===0){

    popup("No workout data to export.");

    return;

  }

  try{

    const csv = makeCSV();

    const blob = new Blob([csv], {type:"text/csv;charset=utf-8"});

    const url = URL.createObjectURL(blob);

    const a = document.createElement("a");

    a.href = url;

    a.download = "SavFit_Workouts.csv";

    document.body.appendChild(a);

    a.click();

    setTimeout(()=>{

      document.body.removeChild(a);

      URL.revokeObjectURL(url);

    },100);

    popup("CSV export created. If your phone blocks downloads, use Copy CSV instead.");

  }catch(e){

    popup("CSV download failed. Use Copy CSV instead.");

  }

}
